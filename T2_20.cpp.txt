#include <iostream>

using namespace std;

int main()
{
     int r,pi=3.14159;
     cin>>r;
     cout<<"面积为"<<pi*r*r<<endl;
     cout<<"周长为"<<2*pi*r<<endl;
     cout<<"直径为"<<2*r<<endl;
    return 0;
}
